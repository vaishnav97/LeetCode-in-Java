class Solution {
    public boolean containsDuplicate(int[] nums) {
        if (nums == null || nums.length == 0) {
            return false;
        }

        Map<Integer, Integer> visited = new HashMap<>();
        for (int num : nums) {
            if (visited.containsKey(num)) {
                return true;
            }

            visited.put(num, visited.getOrDefault(num, 0) + 1);
        }

        return false;
    }
}
