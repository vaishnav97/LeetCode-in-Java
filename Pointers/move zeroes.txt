class Solution {
    private void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }

    public void moveZeroes(int[] nums) {
        int index = 0, j = 0;

        while (j < nums.length) {
            if (nums[j] != 0) {
                swap(nums, index, j);
                index++;
            }

            j++;
        }
    }
}
