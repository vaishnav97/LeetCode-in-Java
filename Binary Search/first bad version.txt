public class Solution extends VersionControl {
    public int firstBadVersion(int n) {
        int low = 1, high = n;

        while (low <= high) {
            int mid = low + (high - low) / 2;

            boolean isBad = isBadVersion(mid);

            if (isBad) {
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }

        return low;
    }
}
