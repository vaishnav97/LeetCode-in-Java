public class Solution extends VersionControl {
    public int firstBadVersion(int n) {
        int start = 1, end = n;
        int ans = -1;

        while (start <= end) {
            int mid = start + (end - start) / 2;

            boolean isBad = isBadVersion(mid);

            if (isBad) {
                ans = mid;
                end = mid - 1;
            } else {
                start = mid + 1;
            }
        }

        return ans;
    }
}
